   ```erlang
   emit_to_logger(GleamLevel, LogRecord) ->
       Level = gleam_level_to_atom(GleamLevel),
       Message = LogRecord#log_record.message,
       logger:log(Level, "~ts", [Message], #{

       }),

   ```

3. Also update `build_log_record_from_otp/3` which constructs a LogRecord tuple directly. Currently it uses:
   ```erlang
   {log_record, Timestamp, GleamLevel, LoggerName, Message, Metadata, none}.
   ```
   Replace with record syntax:
   ```erlang
   #log_record{
       timestamp = Timestamp,
       level = GleamLevel,
       logger_name = LoggerName,
       message = Message,
       metadata = Metadata,
       caller_id = none
   }.
   ```

The `-include` directive causes the Erlang compiler to catch any field reordering at compile time instead of producing silent runtime corruption.

Do NOT change the `-spec` annotations for `emit_to_logger/2` -- they use `tuple()` which is correct since the Gleam type is opaque to Erlang's type system. The record syntax provides compile-time safety through the header, not through typespecs.

Add the implementation:
```erlang
%% Check if birch's :logger integration is healthy.
%% Unlike ensure_initialized/0 (which checks persistent_term cache),
%% this queries the actual :logger handler state.
%% Use for health checks and monitoring, not in hot paths.
-spec is_healthy() -> boolean().
is_healthy() ->
    case logger:get_handler_config(default) of
        {ok, #{formatter := {birch_erlang_logger_ffi, #{format_fn := _}}}} ->
            true;
        _ ->
            false
    end.
```

Note: This is intentionally more specific than `is_formatter_configured/0` -- it checks that `format_fn` is present in the config map, confirming birch's formatter (not just any module named birch_erlang_logger_ffi) is active.

pub fn is_healthy_returns_true_when_formatter_installed_test() {
  case is_erlang_target() {
    True -> {
      // Install formatter
      let assert Ok(Nil) = erlang_logger.setup()

      // Should be healthy
      erlang_logger.is_healthy()
      |> should.be_true

      // Cleanup
      let _ = erlang_logger.remove_formatter()
      Nil
    }
    False -> {
      // On JS, should always return false
      erlang_logger.is_healthy()
      |> should.be_false
    }
  }
}

pub fn is_healthy_returns_false_when_formatter_removed_test() {
  case is_erlang_target() {

      // Should NOT be healthy even though persistent_term may be stale
      erlang_logger.is_healthy()

    False -> Nil

  }
